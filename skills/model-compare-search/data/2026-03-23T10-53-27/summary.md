  📱 手势冲突解决方案（推荐实现）

  // 来自 GLM-5 + Kimi 的综合方案
  handleTouchMove(e) {
    const { scrollTop } = this.list;
    const isAtTop = scrollTop <= 0;
    const isPullingDown = e.deltaY > 0;
    const isFastSwipe = e.velocity > 1.5; // Fling gesture

    // 快速滑动直接关闭
    if (isFastSwipe) {
      this.close();
      return;
    }

    // 顶部下拉才关闭面板
    if (isAtTop && isPullingDown) {
      this.startPanelDrag(e);
    } else {
      // 否则只滚动列表
      this.allowListScroll(e);
    }
  }
